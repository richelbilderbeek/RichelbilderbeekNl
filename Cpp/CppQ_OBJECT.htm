(C++) Q_OBJECT

 

Q_OBJECT is a Qt macro that every class derived from QObject that uses signals and slots must have in its declaration.

 

 

 

 

 

emitter.h

 

#ifndef EMITTER_H
#define EMITTER_H

#include <QObject>

struct Emitter : public QObject
{
  Q_OBJECT
  public:
    void emit_ok();
  signals:
    void ok();
};

#endif // EMITTER_H

 

 

 

 

 

emitter.cpp

 

#include <iostream>
#include "emitter.h"

void Emitter::emit_ok()
{
  std::clog << "Emitting\n";
  emit ok();
}
