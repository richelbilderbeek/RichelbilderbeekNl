(C++) CliToInt

 

CliToInt is a cln::cl_I conversion code snippets to convert a cln::cl_I to an int.

 

 

#include <cln/cln.h>

///CliToInt converts a cln::cl_I to int.
///From http://www.richelbilderbeek.nl/CppCliToInt.htm
int CliToInt(const cln::cl_I& i)
{
  return cln::cl_I_to_int(i);
}
