(C++) #ifdef

 

#ifdef is an abbreviation of 'if defined' and is the preprocessor equivalent of an if-statement.

 

#ifdef HIGH_FRAME_RATE
  const int frames_per_second = 60;
#else
  const int frames_per_second = 20;
#endif
