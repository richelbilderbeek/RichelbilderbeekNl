#include <cassert>
#include <iostream>
#include <random>

int main()
{
  const double lambda{1.0};
  std::exponential_distribution<double> d(lambda);
  assert(lambda == d.lambda());

  const int rng_seed{42};
  std::mt19937 rng(rng_seed);

  for (int i=0; i!=20; ++i)
  {
    std::cout << d(rng) << '\n';
  }
}
