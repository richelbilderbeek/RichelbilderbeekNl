(C++) #undef

 

#undef is a preprocessor directive to undo a #define.

 

#include <iostream>

#define MY_DEFINE

int main()
{
  #ifdef MY_DEFINE
  std::cout << "MY_DEFINE is #defined\n";
  #else
  std::cout << "MY_DEFINE is not #defined\n";
  #endif

  #undef MY_DEFINE

  #ifdef MY_DEFINE
  std::cout << "MY_DEFINE is #defined\n";
  #else
  std::cout << "MY_DEFINE is not #defined\n";
  #endif
}

 

Screen output:

 

MY_DEFINE is #defined
MY_DEFINE is not #define
